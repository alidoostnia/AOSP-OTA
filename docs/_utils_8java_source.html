96 public static boolean isCompatible(UpdateBaseInfo update) {
97 if (update.getVersion().compareTo(SystemProperties.get(Constants.PROP_BUILD_VERSION)) < 0) {
98 Log.d(TAG, update.getName() + " is older than current Android version");
99 return false;
100 }
101 if (!SystemProperties.getBoolean(Constants.PROP_UPDATER_ALLOW_DOWNGRADING, false) &&
102 update.getTimestamp() <= SystemProperties.getLong(Constants.PROP_BUILD_DATE, 0)) {
103 Log.d(TAG, update.getName() + " is older than/equal to the current build");
104 return false;
105 }
106 if (!update.getType().equalsIgnoreCase(SystemProperties.get(Constants.PROP_RELEASE_TYPE))) {
107 Log.d(TAG, update.getName() + " has type " + update.getType());
108 return false;
109 }
110 return true;
111 }

128 public static List<UpdateInfo> parseJson(File file, boolean compatibleOnly)
129 throws IOException, JSONException {
130 List<UpdateInfo> updates = new ArrayList<>();
131
132 StringBuilder json = new StringBuilder();
133 try (BufferedReader br = new BufferedReader(new FileReader(file))) {
134 for (String line; (line = br.readLine()) != null;) {
135 json.append(line);
136 }
137 }
138
139 JSONObject obj = new JSONObject(json.toString());
140 JSONArray updatesList = obj.getJSONArray("response");
141 for (int i = 0; i < updatesList.length(); i++) {
142 if (updatesList.isNull(i)) {
143 continue;
144 }
145 try {
146 UpdateInfo update = parseJsonUpdate(updatesList.getJSONObject(i));
147 if (!compatibleOnly || isCompatible(update)) {
148 updates.add(update);
149 } else {
150 Log.d(TAG, "Ignoring incompatible update " + update.getName());
151 }
152 } catch (JSONException e) {
153 Log.e(TAG, "Could not parse update object, index=" + i, e);
154 }
155 }
156
157 return updates;
158 }

215 public static boolean checkForNewUpdates(File oldJson, File newJson)
216 throws IOException, JSONException {
217 List<UpdateInfo> oldList = parseJson(oldJson, true);
218 List<UpdateInfo> newList = parseJson(newJson, true);
219 Set<String> oldIds = new HashSet<>();
220 for (UpdateInfo update : oldList) {
221 oldIds.add(update.getDownloadId());
222 }
223 // In case of no new updates, the old list should
224 // have all (if not more) the updates
225 for (UpdateInfo update : newList) {
226 if (!oldIds.contains(update.getDownloadId())) {
227 return true;
228 }
229 }
230 return false;
231 }

241 public static long getZipEntryOffset(ZipFile zipFile, String entryPath) {
242 // Each entry has an header of (30 + n + m) bytes
243 // 'n' is the length of the file name
244 // 'm' is the length of the extra field
245 final int FIXED_HEADER_SIZE = 30;
246 Enumeration<? extends ZipEntry> zipEntries = zipFile.entries();
247 long offset = 0;
248 while (zipEntries.hasMoreElements()) {
249 ZipEntry entry = zipEntries.nextElement();
250 int n = entry.getName().length();
251 int m = entry.getExtra() == null ? 0 : entry.getExtra().length;
252 int headerSize = FIXED_HEADER_SIZE + n + m;
253 offset += headerSize;
254 if (entry.getName().equals(entryPath)) {
255 return offset;
256 }
257 offset += entry.getCompressedSize();
258 }
259 Log.e(TAG, "Entry " + entryPath + " not found");
260 throw new IllegalArgumentException("The given entry was not found");
261 }

281 public static void cleanupDownloadsDir(Context context) {
282 File downloadPath = getDownloadPath(context);
283 SharedPreferences preferences = PreferenceManager.getDefaultSharedPreferences(context);
284
285 removeUncryptFiles(downloadPath);
286
287 long buildTimestamp = SystemProperties.getLong(Constants.PROP_BUILD_DATE, 0);
288 long prevTimestamp = preferences.getLong(Constants.PREF_INSTALL_OLD_TIMESTAMP, 0);
289 String lastUpdatePath = preferences.getString(Constants.PREF_INSTALL_PACKAGE_PATH, null);
290 boolean reinstalling = preferences.getBoolean(Constants.PREF_INSTALL_AGAIN, false);
291 boolean deleteUpdates = preferences.getBoolean(Constants.PREF_AUTO_DELETE_UPDATES, false);
292 if ((buildTimestamp != prevTimestamp || reinstalling) && deleteUpdates &&
293 lastUpdatePath != null) {
294 File lastUpdate = new File(lastUpdatePath);
295 if (lastUpdate.exists()) {
296 //noinspection ResultOfMethodCallIgnored
297 lastUpdate.delete();
298 // Remove the pref not to delete the file if re-downloaded
299 preferences.edit().remove(Constants.PREF_INSTALL_PACKAGE_PATH).apply();
300 }
301 }
302
303 final String DOWNLOADS_CLEANUP_DONE = "cleanup_done";
304 if (preferences.getBoolean(DOWNLOADS_CLEANUP_DONE, false)) {
305 return;
306 }
307
308 Log.d(TAG, "Cleaning " + downloadPath);
309 if (!downloadPath.isDirectory()) {
310 return;
311 }
312 File[] files = downloadPath.listFiles();
313 if (files == null) {
314 return;
315 }
316
317 // Ideally the database is empty when we get here
318 UpdatesDbHelper dbHelper = new UpdatesDbHelper(context);
319 List<String> knownPaths = new ArrayList<>();
320 for (UpdateInfo update : dbHelper.getUpdates()) {
321 knownPaths.add(update.getFile().getAbsolutePath());
322 }
323 for (File file : files) {
324 if (!knownPaths.contains(file.getAbsolutePath())) {
325 Log.d(TAG, "Deleting " + file.getAbsolutePath());
326 //noinspection ResultOfMethodCallIgnored
327 file.delete();
328 }
329 }
330
331 preferences.edit().putBoolean(DOWNLOADS_CLEANUP_DONE, true).apply();
332 }

334 public static File appendSequentialNumber(final File file) {
335 String name;
336 String extension;
337 int extensionPosition = file.getName().lastIndexOf(".");
338 if (extensionPosition > 0) {
339 name = file.getName().substring(0, extensionPosition);
340 extension = file.getName().substring(extensionPosition);
341 } else {
342 name = file.getName();
343 extension = "";
344 }
345 final File parent = file.getParentFile();
346 for (int i = 1; i < Integer.MAX_VALUE; i++) {
347 File newFile = new File(parent, name + "-" + i + extension);
348 if (!newFile.exists()) {
349 return newFile;
350 }
351 }
352 throw new IllegalStateException();
353 }
