143 private void calculateSpeed(boolean justResumed) {
144 final long millis = SystemClock.elapsedRealtime();
145 if (justResumed) {
146 // If we don't start over with these after resumption, we get huge numbers for
147 // ETA since the delta will grow, resulting in a very low speed
148 mLastMillis = millis;
149 mSpeed = -1; // we don't want the moving avg with values from who knows when
150
151 // need to do this as well, otherwise the second time we call calculateSpeed(),
152 // the difference (mTotalBytesRead - mCurSampleBytes) will be larger than expected,
153 // resulting in a higher speed calculation
154 mCurSampleBytes = mTotalBytesRead;
155 return;
156 }
157 final long delta = millis - mLastMillis;
158 if (delta > 500) {
159 final long curSpeed = ((mTotalBytesRead - mCurSampleBytes) * 1000) / delta;
160 if (mSpeed == -1) {
161 mSpeed = curSpeed;
162 } else {
163 mSpeed = ((mSpeed * 3) + curSpeed) / 4;
164 }
165
166 mLastMillis = millis;
167 mCurSampleBytes = mTotalBytesRead;
168 }
169 }
170
171 private void calculateEta() {
172 if (mSpeed > 0) {
173 mEta = (mTotalBytes - mTotalBytesRead) / mSpeed;
174 }
175 }

186 private void handleDuplicateLinks() throws IOException {
187 String protocol = mClient.getURL().getProtocol();
188
189 class DuplicateLink {
190 private final String mUrl;
191 private final int mPriority;
192 private DuplicateLink(String url, int priority) {
193 mUrl = url;
194 mPriority = priority;
195 }
196 }
197
198 PriorityQueue<DuplicateLink> duplicates = null;
199
200 for (Map.Entry<String, List<String>> entry : mClient.getHeaderFields().entrySet()) {
201 if ("Link".equalsIgnoreCase((entry.getKey()))) {
202 duplicates = new PriorityQueue<>(entry.getValue().size(),
203 Comparator.comparingInt(d -> d.mPriority));
204
205 // https://tools.ietf.org/html/rfc6249
206 // https://tools.ietf.org/html/rfc5988#section-5
207 String regex = "(?i)<(.+)>\\s*;\\s*rel=duplicate(?:.*pri=([0-9]+).*|.*)?";
208 Pattern pattern = Pattern.compile(regex);
209 for (String field : entry.getValue()) {
210 Matcher matcher = pattern.matcher(field);
211 if (matcher.matches()) {
212 String url = matcher.group(1);
213 String pri = matcher.group(2);
214 int priority = pri != null ? Integer.parseInt(pri) : 999999;
215 duplicates.add(new DuplicateLink(url, priority));
216 Log.d(TAG, "Adding duplicate link " + url);
217 } else {
218 Log.d(TAG, "Ignoring link " + field);
219 }
220 }
221 }
222 }
223
224 String newUrl = mClient.getHeaderField("Location");
225 for (;;) {
226 try {
227 URL url = new URL(newUrl);
228 if (!url.getProtocol().equals(protocol)) {
229 // If we hadn't handled duplicate links, we wouldn't have
230 // used this url.
231 throw new IOException("Protocol changes are not allowed");
232 }
233 Log.d(TAG, "Downloading from " + newUrl);
234 changeClientUrl(url);
235 mClient.setConnectTimeout(5000);
236 mClient.connect();
237 if (!isSuccessCode(mClient.getResponseCode())) {
238 throw new IOException("Server replied with " + mClient.getResponseCode());
239 }
240 return;
241 } catch (IOException e) {
242 if (duplicates != null && !duplicates.isEmpty()) {
243 DuplicateLink link = duplicates.poll();
244 if (link != null) {
245 duplicates.remove(link);
246 newUrl = link.mUrl;
247 Log.e(TAG, "Using duplicate link " + link.mUrl, e);
248 }
249 } else {
250 throw e;
251 }
252 }
253 }
254 }

256 @Override
257 public void run() {
258 boolean justResumed = false;
259 try {
260 mClient.setInstanceFollowRedirects(!mUseDuplicateLinks);
261 mClient.connect();
262 int responseCode = mClient.getResponseCode();
263
264 if (mUseDuplicateLinks && isRedirectCode(responseCode)) {
265 handleDuplicateLinks();
266 responseCode = mClient.getResponseCode();
267 }
268
269 mCallback.onResponse(new Headers());
270
271 if (mResume && isPartialContentCode(responseCode)) {
272 justResumed = true;
273 mTotalBytesRead = mDestination.length();
274 Log.d(TAG, "The server fulfilled the partial content request");
275 } else if (mResume || !isSuccessCode(responseCode)) {
276 Log.e(TAG, "The server replied with code " + responseCode);
277 mCallback.onFailure(isInterrupted());
278 return;
279 }
280
281 try (
282 InputStream inputStream = mClient.getInputStream();
283 OutputStream outputStream = new FileOutputStream(mDestination, mResume)
284 ) {
285 mTotalBytes = mClient.getContentLength() + mTotalBytesRead;
286 byte[] b = new byte[8192];
287 int count;
288 while (!isInterrupted() && (count = inputStream.read(b)) > 0) {
289 outputStream.write(b, 0, count);
290 mTotalBytesRead += count;
291 calculateSpeed(justResumed);
292 calculateEta();
293 justResumed = false; // otherwise we will never get speed and ETA again
294 if (mProgressListener != null) {
295 mProgressListener.update(mTotalBytesRead, mTotalBytes, mSpeed, mEta);
296 }
297 }
298 if (mProgressListener != null) {
299 mProgressListener.update(mTotalBytesRead, mTotalBytes, mSpeed, mEta);
300 }
301
302 outputStream.flush();
303
304 if (isInterrupted()) {
305 mCallback.onFailure(true);
306 } else {
307 mCallback.onSuccess();
308 }
309 }
310 } catch (IOException e) {
311 Log.e(TAG, "Error downloading file", e);
312 mCallback.onFailure(isInterrupted());
313 } finally {
314 mClient.disconnect();
315 }
316 }
