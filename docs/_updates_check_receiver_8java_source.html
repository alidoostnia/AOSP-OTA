52 @Override
53 public void onReceive(final Context context, Intent intent) {
54 if (Intent.ACTION_BOOT_COMPLETED.equals(intent.getAction())) {
55 Utils.cleanupDownloadsDir(context);
56 }
57
58 final SharedPreferences preferences =
59 PreferenceManager.getDefaultSharedPreferences(context);
60
61 if (!Utils.isUpdateCheckEnabled(context)) {
62 return;
63 }
64
65 if (Intent.ACTION_BOOT_COMPLETED.equals(intent.getAction())) {
66 // Set a repeating alarm on boot to check for new updates once per day
67 scheduleRepeatingUpdatesCheck(context);
68 }
69
70 if (!Utils.isNetworkAvailable(context)) {
71 Log.d(TAG, "Network not available, scheduling new check");
72 scheduleUpdatesCheck(context);
73 return;
74 }
75
76 final File json = Utils.getCachedUpdateList(context);
77 final File jsonNew = new File(json.getAbsolutePath() + UUID.randomUUID());
78 String url = Utils.getServerURL(context);
79 DownloadClient.DownloadCallback callback = new DownloadClient.DownloadCallback() {
80 @Override
81 public void onFailure(boolean cancelled) {
82 Log.e(TAG, "Could not download updates list, scheduling new check");
83 scheduleUpdatesCheck(context);
84 }
85
86 @Override
87 public void onResponse(DownloadClient.Headers headers) {
88 }
89
90 @Override
91 public void onSuccess() {
92 try {
93 if (json.exists() && Utils.checkForNewUpdates(json, jsonNew)) {
94 showNotification(context);
95 updateRepeatingUpdatesCheck(context);
96 }
97 //noinspection ResultOfMethodCallIgnored
98 jsonNew.renameTo(json);
99 long currentMillis = System.currentTimeMillis();
100 preferences.edit()
101 .putLong(Constants.PREF_LAST_UPDATE_CHECK, currentMillis)
102 .apply();
103 // In case we set a one-shot check because of a previous failure
104 cancelUpdatesCheck(context);
105 } catch (IOException | JSONException e) {
106 Log.e(TAG, "Could not parse list, scheduling new check", e);
107 scheduleUpdatesCheck(context);
108 }
109 }
110 };
111
112 try {
113 DownloadClient downloadClient = new DownloadClient.Builder()
114 .setUrl(url)
115 .setDestination(jsonNew)
116 .setDownloadCallback(callback)
117 .build();
118 downloadClient.start();
119 } catch (IOException e) {
120 Log.e(TAG, "Could not fetch list, scheduling new check", e);
121 scheduleUpdatesCheck(context);
122 }
123 }

125 private static void showNotification(Context context) {
126 NotificationManager notificationManager = context.getSystemService(
127 NotificationManager.class);
128 NotificationChannel notificationChannel = new NotificationChannel(
129 NEW_UPDATES_NOTIFICATION_CHANNEL,
130 context.getString(R.string.new_updates_channel_title),
131 NotificationManager.IMPORTANCE_LOW);
132 NotificationCompat.Builder notificationBuilder = new NotificationCompat.Builder(context,
133 NEW_UPDATES_NOTIFICATION_CHANNEL);
134 notificationBuilder.setSmallIcon(R.drawable.ic_system_update);
135 Intent notificationIntent = new Intent(context, UpdatesActivity.class);
136 PendingIntent intent = PendingIntent.getActivity(context, 0, notificationIntent,
137 PendingIntent.FLAG_UPDATE_CURRENT | PendingIntent.FLAG_IMMUTABLE);
138 notificationBuilder.setContentIntent(intent);
139 notificationBuilder.setContentTitle(context.getString(R.string.new_updates_found_title));
140 notificationBuilder.setAutoCancel(true);
141 notificationManager.createNotificationChannel(notificationChannel);
142 notificationManager.notify(0, notificationBuilder.build());
143 }

152 cancelRepeatingUpdatesCheck(context);
153 scheduleRepeatingUpdatesCheck(context);
154 }

156 public static void scheduleRepeatingUpdatesCheck(Context context) {
157 if (!Utils.isUpdateCheckEnabled(context)) {
158 return;
159 }
160
161 PendingIntent updateCheckIntent = getRepeatingUpdatesCheckIntent(context);
162 AlarmManager alarmMgr = context.getSystemService(AlarmManager.class);
163 alarmMgr.setRepeating(AlarmManager.RTC, System.currentTimeMillis() +
164 Utils.getUpdateCheckInterval(context), Utils.getUpdateCheckInterval(context),
165 updateCheckIntent);
166
167 Date nextCheckDate = new Date(System.currentTimeMillis() +
168 Utils.getUpdateCheckInterval(context));
169 Log.d(TAG, "Setting automatic updates check: " + nextCheckDate);
170 }
171
172 public static void cancelRepeatingUpdatesCheck(Context context) {
173 AlarmManager alarmMgr = context.getSystemService(AlarmManager.class);
174 alarmMgr.cancel(getRepeatingUpdatesCheckIntent(context));
175 }

177 private static PendingIntent getUpdatesCheckIntent(Context context) {
178 Intent intent = new Intent(context, UpdatesCheckReceiver.class);
179 intent.setAction(ONESHOT_CHECK_ACTION);
180 return PendingIntent.getBroadcast(context, 0, intent, PendingIntent.FLAG_IMMUTABLE);
181 }
182
183 public static void scheduleUpdatesCheck(Context context) {
184 long millisToNextCheck = AlarmManager.INTERVAL_HOUR * 2;
185 PendingIntent updateCheckIntent = getUpdatesCheckIntent(context);
186 AlarmManager alarmMgr = context.getSystemService(AlarmManager.class);
187 alarmMgr.set(AlarmManager.ELAPSED_REALTIME,
188 SystemClock.elapsedRealtime() + millisToNextCheck,
189 updateCheckIntent);
190
191 Date nextCheckDate = new Date(System.currentTimeMillis() + millisToNextCheck);
192 Log.d(TAG, "Setting one-shot updates check: " + nextCheckDate);
193 }
194
195 public static void cancelUpdatesCheck(Context context) {
196 AlarmManager alarmMgr = context.getSystemService(AlarmManager.class);
197 alarmMgr.cancel(getUpdatesCheckIntent(context));
198 Log.d(TAG, "Cancelling pending one-shot check");
199 }
