39 private static final String TAG = "ExportUpdateService";

41 private static final int NOTIFICATION_ID = 16;

43 public static final String ACTION_START_EXPORTING = "start_exporting";

45 public static final String EXTRA_SOURCE_FILE = "source_file";
46 public static final String EXTRA_DEST_URI = "dest_uri";
47
48 private static final String EXPORT_NOTIFICATION_CHANNEL =
49 "export_notification_channel";
50
51 private volatile boolean mIsExporting = false;
52
53 private Thread mExportThread;

55 @Override
56 public IBinder onBind(Intent intent) {
57 return null;
58 }

60 @Override
61 public int onStartCommand(Intent intent, int flags, int startId) {
62 if (ACTION_START_EXPORTING.equals(intent.getAction())) {
63 if (mIsExporting) {
64 Log.e(TAG, "Already exporting an update");
65 Toast.makeText(this, R.string.toast_already_exporting, Toast.LENGTH_SHORT).show();
66 return START_NOT_STICKY;
67 }
68 mIsExporting = true;
69 File source = (File) intent.getSerializableExtra(EXTRA_SOURCE_FILE);
70 Uri destination = intent.getParcelableExtra(EXTRA_DEST_URI);
71 startExporting(source, destination);
72 Toast.makeText(this, R.string.toast_export_started, Toast.LENGTH_SHORT).show();
73 } else {
74 Log.e(TAG, "No action specified");
75 }
76
77 if (!mIsExporting) {
78 stopSelf();
79 }
80
81 return START_NOT_STICKY;
82 }

84 private class ExportRunnable implements Runnable {
85 private final ContentResolver mContentResolver;
86 private final File mSource;
87 private final Uri mDestination;
88 private final FileUtils.ProgressCallBack mProgressCallBack;
89 private final Runnable mRunnableComplete;
90 private final Runnable mRunnableFailed;
91
92 private ExportRunnable(ContentResolver cr, File source, Uri destination,
93 FileUtils.ProgressCallBack progressCallBack,
94 Runnable runnableComplete, Runnable runnableFailed) {
95 mContentResolver = cr;
96 mSource = source;
97 mDestination = destination;
98 mProgressCallBack = progressCallBack;
99 mRunnableComplete = runnableComplete;
100 mRunnableFailed = runnableFailed;
101 }
102
103 @Override
104 public void run() {
105 try {
106 FileUtils.copyFile(mContentResolver, mSource, mDestination, mProgressCallBack);
107 mIsExporting = false;
108 if (!mExportThread.isInterrupted()) {
109 Log.d(TAG, "Completed");
110 mRunnableComplete.run();
111 } else {
112 Log.d(TAG, "Aborted");
113 }
114 } catch (IOException e) {
115 mIsExporting = false;
116 Log.e(TAG, "Could not copy file", e);
117 mRunnableFailed.run();
118 } finally {
119 stopSelf();
120 }
121 }
122 }

124 private void startExporting(File source, Uri destination) {
125 final String fileName = FileUtils.queryName(getContentResolver(), destination);
126 NotificationManager notificationManager = getSystemService(NotificationManager.class);
127 NotificationChannel notificationChannel = new NotificationChannel(
128 EXPORT_NOTIFICATION_CHANNEL,
129 getString(R.string.export_channel_title),
130 NotificationManager.IMPORTANCE_LOW);
131 notificationManager.createNotificationChannel(notificationChannel);
132
133 NotificationCompat.Builder notificationBuilder = new NotificationCompat.Builder(this,
134 EXPORT_NOTIFICATION_CHANNEL);
135 NotificationCompat.BigTextStyle notificationStyle = new NotificationCompat.BigTextStyle();
136 notificationBuilder.setContentTitle(getString(R.string.dialog_export_title));
137 notificationStyle.setBigContentTitle(getString(R.string.dialog_export_title));
138 notificationStyle.bigText(fileName);
139 notificationBuilder.setStyle(notificationStyle);
140 notificationBuilder.setSmallIcon(R.drawable.ic_system_update);
141
142 FileUtils.ProgressCallBack progressCallBack = new FileUtils.ProgressCallBack() {
143 private long mLastUpdate = -1;
144
145 @Override
146 public void update(int progress) {
147 long now = SystemClock.elapsedRealtime();
148 if (mLastUpdate < 0 || now - mLastUpdate > 500) {
149 String percent = NumberFormat.getPercentInstance().format(progress / 100.f);
150 notificationStyle.setSummaryText(percent);
151 notificationBuilder.setProgress(100, progress, false);
152 notificationManager.notify(NOTIFICATION_ID,
153 notificationBuilder.build());
154 mLastUpdate = now;
155 }
156 }
157 };
158
159 startForeground(NOTIFICATION_ID, notificationBuilder.build());
160 notificationManager.notify(NOTIFICATION_ID, notificationBuilder.build());
161
162 Runnable runnableComplete = () -> {
163 notificationStyle.setSummaryText(null);
164 notificationStyle.setBigContentTitle(
165 getString(R.string.notification_export_success));
166 notificationBuilder.setContentTitle(
167 getString(R.string.notification_export_success));
168 notificationBuilder.setProgress(0, 0, false);
169 notificationBuilder.setContentText(fileName);
170 notificationManager.notify(NOTIFICATION_ID, notificationBuilder.build());
171 stopForeground(STOP_FOREGROUND_DETACH);
172 };
173
174 Runnable runnableFailed = () -> {
175 notificationStyle.setSummaryText(null);
176 notificationStyle.setBigContentTitle(
177 getString(R.string.notification_export_fail));
178 notificationBuilder.setContentTitle(
179 getString(R.string.notification_export_fail));
180 notificationBuilder.setProgress(0, 0, false);
181 notificationBuilder.setContentText(null);
182 notificationManager.notify(NOTIFICATION_ID, notificationBuilder.build());
183 stopForeground(STOP_FOREGROUND_DETACH);
184 };
185
186 ExportRunnable exportRunnable = new ExportRunnable(getContentResolver(), source,
187 destination, progressCallBack, runnableComplete, runnableFailed);
188 mExportThread = new Thread(exportRunnable);
189 mExportThread.start();
190 }
