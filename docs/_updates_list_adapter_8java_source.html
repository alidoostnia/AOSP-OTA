157 private void handleActiveStatus(ViewHolder viewHolder, UpdateInfo update) {
158 boolean canDelete = false;
159
160 final String downloadId = update.getDownloadId();
161 if (mUpdaterController.isDownloading(downloadId)) {
162 canDelete = true;
163 String downloaded = Formatter.formatShortFileSize(mActivity,
164 update.getFile().length());
165 String total = Formatter.formatShortFileSize(mActivity, update.getFileSize());
166 String percentage = NumberFormat.getPercentInstance().format(
167 update.getProgress() / 100.f);
168 viewHolder.mPercentage.setText(percentage);
169 long eta = update.getEta();
170 if (eta > 0) {
171 CharSequence etaString = StringGenerator.formatETA(mActivity, eta * 1000);
172 viewHolder.mProgressText.setText(mActivity.getString(
173 R.string.list_download_progress_eta_newer, downloaded, total, etaString));
174 } else {
175 viewHolder.mProgressText.setText(mActivity.getString(
176 R.string.list_download_progress_newer, downloaded, total));
177 }
178 setButtonAction(viewHolder.mAction, Action.PAUSE, downloadId, true);
179 viewHolder.mProgressBar.setIndeterminate(update.getStatus() == UpdateStatus.STARTING);
180 viewHolder.mProgressBar.setProgress(update.getProgress());
181 } else if (mUpdaterController.isInstallingUpdate(downloadId)) {
182 setButtonAction(viewHolder.mAction, Action.CANCEL_INSTALLATION, downloadId, true);
183 boolean notAB = !mUpdaterController.isInstallingABUpdate();
184 viewHolder.mProgressText.setText(notAB ? R.string.dialog_prepare_zip_message :
185 update.getFinalizing() ?
186 R.string.finalizing_package :
187 R.string.preparing_ota_first_boot);
188 viewHolder.mProgressBar.setIndeterminate(false);
189 viewHolder.mProgressBar.setProgress(update.getInstallProgress());
190 } else if (mUpdaterController.isVerifyingUpdate(downloadId)) {
191 setButtonAction(viewHolder.mAction, Action.INSTALL, downloadId, false);
192 viewHolder.mProgressText.setText(R.string.list_verifying_update);
193 viewHolder.mProgressBar.setIndeterminate(true);
194 } else {
195 canDelete = true;
196 setButtonAction(viewHolder.mAction, Action.RESUME, downloadId, !isBusy());
197 String downloaded = Formatter.formatShortFileSize(mActivity,
198 update.getFile().length());
199 String total = Formatter.formatShortFileSize(mActivity, update.getFileSize());
200 String percentage = NumberFormat.getPercentInstance().format(
201 update.getProgress() / 100.f);
202 viewHolder.mPercentage.setText(percentage);
203 viewHolder.mProgressText.setText(mActivity.getString(
204 R.string.list_download_progress_newer, downloaded, total));
205 viewHolder.mProgressBar.setIndeterminate(false);
206 viewHolder.mProgressBar.setProgress(update.getProgress());
207 }
208
209 viewHolder.mMenu.setOnClickListener(getClickListener(update, canDelete, viewHolder.mMenu));
210 viewHolder.mProgress.setVisibility(View.VISIBLE);
211 viewHolder.mProgressText.setVisibility(View.VISIBLE);
212 viewHolder.mBuildSize.setVisibility(View.INVISIBLE);
213 }

215 private void handleNotActiveStatus(ViewHolder viewHolder, UpdateInfo update) {
216 final String downloadId = update.getDownloadId();
217 if (mUpdaterController.isWaitingForReboot(downloadId)) {
218 viewHolder.mMenu.setOnClickListener(getClickListener(update, false, viewHolder.mMenu));
219 setButtonAction(viewHolder.mAction, Action.REBOOT, downloadId, true);
220 } else if (update.getPersistentStatus() == UpdateStatus.Persistent.VERIFIED) {
221 viewHolder.mMenu.setOnClickListener(getClickListener(update, true, viewHolder.mMenu));
222 setButtonAction(viewHolder.mAction,
223 Utils.canInstall(update) ? Action.INSTALL : Action.DELETE,
224 downloadId, !isBusy());
225 } else if (!Utils.canInstall(update)) {
226 viewHolder.mMenu.setOnClickListener(getClickListener(update, false, viewHolder.mMenu));
227 setButtonAction(viewHolder.mAction, Action.INFO, downloadId, !isBusy());
228 } else {
229 viewHolder.mMenu.setOnClickListener(getClickListener(update, false, viewHolder.mMenu));
230 setButtonAction(viewHolder.mAction, Action.DOWNLOAD, downloadId, !isBusy());
231 }
232 String fileSize = Formatter.formatShortFileSize(mActivity, update.getFileSize());
233 viewHolder.mBuildSize.setText(fileSize);
234
235 viewHolder.mProgress.setVisibility(View.INVISIBLE);
236 viewHolder.mProgressText.setVisibility(View.INVISIBLE);
237 viewHolder.mBuildSize.setVisibility(View.VISIBLE);
238 }

240 @Override
241 public void onBindViewHolder(@NonNull final ViewHolder viewHolder, int i) {
242 if (mDownloadIds == null) {
243 viewHolder.mAction.setEnabled(false);
244 return;
245 }
246
247 final String downloadId = mDownloadIds.get(i);
248 UpdateInfo update = mUpdaterController.getUpdate(downloadId);
249 if (update == null) {
250 // The update was deleted
251 viewHolder.mAction.setEnabled(false);
252 viewHolder.mAction.setText(R.string.action_download);
253 return;
254 }
255
256 viewHolder.itemView.setSelected(downloadId.equals(mSelectedDownload));
257
258 boolean activeLayout;
259 switch (update.getPersistentStatus()) {
260 case UpdateStatus.Persistent.UNKNOWN:
261 activeLayout = update.getStatus() == UpdateStatus.STARTING;
262 break;
263 case UpdateStatus.Persistent.VERIFIED:
264 activeLayout = update.getStatus() == UpdateStatus.INSTALLING;
265 break;
266 case UpdateStatus.Persistent.INCOMPLETE:
267 activeLayout = true;
268 break;
269 default:
270 throw new RuntimeException("Unknown update status");
271 }
272
273 String buildDate = StringGenerator.getDateLocalizedUTC(mActivity,
274 DateFormat.LONG, update.getTimestamp());
275 String buildVersion = mActivity.getString(R.string.list_build_version,
276 update.getVersion());
277 viewHolder.mBuildDate.setText(buildDate);
278 viewHolder.mBuildVersion.setText(buildVersion);
279 viewHolder.mBuildVersion.setCompoundDrawables(null, null, null, null);
280
281 if (activeLayout) {
282 handleActiveStatus(viewHolder, update);
283 } else {
284 handleNotActiveStatus(viewHolder, update);
285 }
286 }

344 private void setButtonAction(Button button, Action action, final String downloadId,
345 boolean enabled) {
346 final View.OnClickListener clickListener;
347 switch (action) {
348 case DOWNLOAD:
349 button.setText(R.string.action_download);
350 button.setEnabled(enabled);
351 clickListener = enabled ? view -> startDownloadWithWarning(downloadId) : null;
352 break;
353 case PAUSE:
354 button.setText(R.string.action_pause);
355 button.setEnabled(enabled);
356 clickListener = enabled ? view -> mUpdaterController.pauseDownload(downloadId)
357 : null;
358 break;
359 case RESUME: {
360 button.setText(R.string.action_resume);
361 button.setEnabled(enabled);
362 UpdateInfo update = mUpdaterController.getUpdate(downloadId);
363 final boolean canInstall = Utils.canInstall(update) ||
364 update.getFile().length() == update.getFileSize();
365 clickListener = enabled ? view -> {
366 if (canInstall) {
367 mUpdaterController.resumeDownload(downloadId);
368 } else {
369 mActivity.showSnackbar(R.string.snack_update_not_installable,
370 Snackbar.LENGTH_LONG);
371 }
372 } : null;
373 }
374 break;
375 case INSTALL: {
376 button.setText(R.string.action_install);
377 button.setEnabled(enabled);
378 UpdateInfo update = mUpdaterController.getUpdate(downloadId);
379 final boolean canInstall = Utils.canInstall(update);
380 clickListener = enabled ? view -> {
381 if (canInstall) {
382 AlertDialog.Builder installDialog = getInstallDialog(downloadId);
383 if (installDialog != null) {
384 installDialog.show();
385 }
386 } else {
387 mActivity.showSnackbar(R.string.snack_update_not_installable,
388 Snackbar.LENGTH_LONG);
389 }
390 } : null;
391 }
392 break;
393 case INFO: {
394 button.setText(R.string.action_info);
395 button.setEnabled(enabled);
396 clickListener = enabled ? view -> showInfoDialog() : null;
397 }
398 break;
399 case DELETE: {
400 button.setText(R.string.action_delete);
401 button.setEnabled(enabled);
402 clickListener = enabled ? view -> getDeleteDialog(downloadId).show() : null;
403 }
404 break;
405 case CANCEL_INSTALLATION: {
406 button.setText(R.string.action_cancel);
407 button.setEnabled(enabled);
408 clickListener = enabled ? view -> getCancelInstallationDialog().show() : null;
409 }
410 break;
411 case REBOOT: {
412 button.setText(R.string.reboot);
413 button.setEnabled(enabled);
414 clickListener = enabled ? view -> {
415 PowerManager pm = mActivity.getSystemService(PowerManager.class);
416 pm.reboot(null);
417 } : null;
418 }
419 break;
420 default:
421 clickListener = null;
422 }
423 button.setAlpha(enabled ? 1.f : mAlphaDisabledValue);
424
425 // Disable action mode when a button is clicked
426 button.setOnClickListener(v -> {
427 if (clickListener != null) {
428 clickListener.onClick(v);
429 }
430 });
431 }

522 private void startActionMode(final UpdateInfo update, final boolean canDelete, View anchor) {
523 mSelectedDownload = update.getDownloadId();
524 notifyItemChanged(update.getDownloadId());
525
526 ContextThemeWrapper wrapper = new ContextThemeWrapper(mActivity,
527 R.style.AppTheme_PopupMenuOverlapAnchor);
528 PopupMenu popupMenu = new PopupMenu(wrapper, anchor, Gravity.NO_GRAVITY,
529 R.attr.actionOverflowMenuStyle, 0);
530 popupMenu.inflate(R.menu.menu_action_mode);
531
532 MenuBuilder menu = (MenuBuilder) popupMenu.getMenu();
533 menu.findItem(R.id.menu_delete_action).setVisible(canDelete);
534 menu.findItem(R.id.menu_copy_url).setVisible(update.getAvailableOnline());
535 menu.findItem(R.id.menu_export_update).setVisible(
536 update.getPersistentStatus() == UpdateStatus.Persistent.VERIFIED);
537
538 popupMenu.setOnMenuItemClickListener(item -> {
539 int itemId = item.getItemId();
540 if (itemId == R.id.menu_delete_action) {
541 getDeleteDialog(update.getDownloadId()).show();
542 return true;
543 } else if (itemId == R.id.menu_copy_url) {
544 Utils.addToClipboard(mActivity,
545 mActivity.getString(R.string.label_download_url),
546 update.getDownloadUrl(),
547 mActivity.getString(R.string.toast_download_url_copied));
548 return true;
549 } else if (itemId == R.id.menu_export_update) {
550 if (mActivity != null) {
551 mActivity.exportUpdate(update);
552 }
553 return true;
554 }
555 return false;
556 });
557
558 MenuPopupHelper helper = new MenuPopupHelper(wrapper, menu, anchor);
559 helper.show();
560 }

582 private boolean isBatteryLevelOk() {
583 Intent intent = mActivity.registerReceiver(null,
584 new IntentFilter(Intent.ACTION_BATTERY_CHANGED));
585 if (!intent.getBooleanExtra(BatteryManager.EXTRA_PRESENT, false)) {
586 return true;
587 }
588 int percent = Math.round(100.f * intent.getIntExtra(BatteryManager.EXTRA_LEVEL, 100) /
589 intent.getIntExtra(BatteryManager.EXTRA_SCALE, 100));
590 int plugged = intent.getIntExtra(BatteryManager.EXTRA_PLUGGED, 0);
591 int required = (plugged & BATTERY_PLUGGED_ANY) != 0 ?
592 mActivity.getResources().getInteger(R.integer.battery_ok_percentage_charging) :
593 mActivity.getResources().getInteger(R.integer.battery_ok_percentage_discharging);
594 return percent >= required;
595 }
