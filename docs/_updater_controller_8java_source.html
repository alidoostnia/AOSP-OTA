152 private DownloadClient.DownloadCallback getDownloadCallback(final String downloadId) {
153 return new DownloadClient.DownloadCallback() {
154
155 @Override
156 public void onResponse(DownloadClient.Headers headers) {
157 final DownloadEntry entry = mDownloads.get(downloadId);
158 if (entry == null) {
159 return;
160 }
161 final Update update = entry.mUpdate;
162 String contentLength = headers.get("Content-Length");
163 if (contentLength != null) {
164 try {
165 long size = Long.parseLong(contentLength);
166 if (update.getFileSize() < size) {
167 update.setFileSize(size);
168 }
169 } catch (NumberFormatException e) {
170 Log.e(TAG, "Could not get content-length");
171 }
172 }
173 update.setStatus(UpdateStatus.DOWNLOADING);
174 update.setPersistentStatus(UpdateStatus.Persistent.INCOMPLETE);
175 new Thread(() -> mUpdatesDbHelper.addUpdateWithOnConflict(update,
176 SQLiteDatabase.CONFLICT_REPLACE)).start();
177 notifyUpdateChange(downloadId);
178 }
179
180 @Override
181 public void onSuccess() {
182 Log.d(TAG, "Download complete");
183 DownloadEntry entry = mDownloads.get(downloadId);
184 if (entry != null) {
185 Update update = entry.mUpdate;
186 update.setStatus(UpdateStatus.VERIFYING);
187 removeDownloadClient(entry);
188 verifyUpdateAsync(downloadId);
189 notifyUpdateChange(downloadId);
190 tryReleaseWakelock();
191 }
192 }
193
194 @Override
195 public void onFailure(boolean cancelled) {
196 if (cancelled) {
197 Log.d(TAG, "Download cancelled");
198 // Already notified
199 } else {
200 DownloadEntry entry = mDownloads.get(downloadId);
201 if (entry != null) {
202 Update update = entry.mUpdate;
203 Log.e(TAG, "Download failed");
204 removeDownloadClient(entry);
205 update.setStatus(UpdateStatus.PAUSED_ERROR);
206 notifyUpdateChange(downloadId);
207 }
208 }
209 tryReleaseWakelock();
210 }
211 };
212 }

214 private DownloadClient.ProgressListener getProgressListener(final String downloadId) {
215 return new DownloadClient.ProgressListener() {
216 private long mLastUpdate = 0;
217 private int mProgress = 0;
218
219 @Override
220 public void update(long bytesRead, long contentLength, long speed, long eta) {
221 DownloadEntry entry = mDownloads.get(downloadId);
222 if (entry == null) {
223 return;
224 }
225 Update update = entry.mUpdate;
226 if (contentLength <= 0) {
227 if (update.getFileSize() <= 0) {
228 return;
229 } else {
230 contentLength = update.getFileSize();
231 }
232 }
233 if (contentLength <= 0) {
234 return;
235 }
236 final long now = SystemClock.elapsedRealtime();
237 int progress = Math.round(bytesRead * 100f / contentLength);
238 if (progress != mProgress || mLastUpdate - now > MAX_REPORT_INTERVAL_MS) {
239 mProgress = progress;
240 mLastUpdate = now;
241 update.setProgress(progress);
242 update.setEta(eta);
243 update.setSpeed(speed);
244 notifyDownloadProgress(downloadId);
245 }
246 }
247 };

250 @SuppressLint("SetWorldReadable")
251 private void verifyUpdateAsync(final String downloadId) {
252 mVerifyingUpdates.add(downloadId);
253 new Thread(() -> {
254 DownloadEntry entry = mDownloads.get(downloadId);
255 if (entry != null) {
256 Update update = entry.mUpdate;
257 File file = update.getFile();
258 if (file.exists() && verifyPackage(file)) {
259 //noinspection ResultOfMethodCallIgnored
260 file.setReadable(true, false);
261 update.setPersistentStatus(UpdateStatus.Persistent.VERIFIED);
262 mUpdatesDbHelper.changeUpdateStatus(update);
263 update.setStatus(UpdateStatus.VERIFIED);
264 } else {
265 update.setPersistentStatus(UpdateStatus.Persistent.UNKNOWN);
266 mUpdatesDbHelper.removeUpdate(downloadId);
267 update.setProgress(0);
268 update.setStatus(UpdateStatus.VERIFICATION_FAILED);
269 }
270 mVerifyingUpdates.remove(downloadId);
271 notifyUpdateChange(downloadId);
272 }
273 }).start();
274 }

276 private boolean verifyPackage(File file) {
277 try {
278 android.os.RecoverySystem.verifyPackage(file, null, null);
279 Log.e(TAG, "Verification successful");
280 return true;
281 } catch (Exception e) {
282 Log.e(TAG, "Verification failed", e);
283 if (file.exists()) {
284 //noinspection ResultOfMethodCallIgnored
285 file.delete();
286 } else {
287 // The download was probably stopped. Exit silently
288 Log.e(TAG, "Error while verifying the file", e);
289 }
290 return false;
291 }

294 private boolean fixUpdateStatus(Update update) {
295 switch (update.getPersistentStatus()) {
296 case UpdateStatus.Persistent.VERIFIED:
297 case UpdateStatus.Persistent.INCOMPLETE:
298 if (update.getFile() == null || !update.getFile().exists()) {
299 update.setStatus(UpdateStatus.UNKNOWN);
300 return false;
301 } else if (update.getFileSize() > 0) {
302 update.setStatus(UpdateStatus.PAUSED);
303 int progress = Math.round(
304 update.getFile().length() * 100f / update.getFileSize());
305 update.setProgress(progress);
306 }
307 break;
308 }
309 return true;
310 }

312 public void setUpdatesAvailableOnline(List<String> downloadIds, boolean purgeList) {
313 List<String> toRemove = new ArrayList<>();
314 for (DownloadEntry entry : mDownloads.values()) {
315 boolean online = downloadIds.contains(entry.mUpdate.getDownloadId());
316 entry.mUpdate.setAvailableOnline(online);
317 if (!online && purgeList &&
318 entry.mUpdate.getPersistentStatus() == UpdateStatus.Persistent.UNKNOWN) {
319 toRemove.add(entry.mUpdate.getDownloadId());
320 }
321 }
322 for (String downloadId : toRemove) {
323 Log.d(TAG, downloadId + " no longer available online, removing");
324 mDownloads.remove(downloadId);
325 notifyUpdateDelete(downloadId);
326 }

333 private boolean addUpdate(final UpdateInfo updateInfo, boolean availableOnline) {
334 Log.d(TAG, "Adding download: " + updateInfo.getDownloadId());
335 if (mDownloads.containsKey(updateInfo.getDownloadId())) {
336 Log.d(TAG, "Download (" + updateInfo.getDownloadId() + ") already added");
337 DownloadEntry entry = mDownloads.get(updateInfo.getDownloadId());
338 if (entry != null) {
339 Update updateAdded = entry.mUpdate;
340 updateAdded.setAvailableOnline(availableOnline && updateAdded.getAvailableOnline());
341 updateAdded.setDownloadUrl(updateInfo.getDownloadUrl());
342 }
343 return false;
344 }
345 Update update = new Update(updateInfo);
346 if (!fixUpdateStatus(update) && !availableOnline) {
347 update.setPersistentStatus(UpdateStatus.Persistent.UNKNOWN);
348 deleteUpdateAsync(update);
349 Log.d(TAG, update.getDownloadId() + " had an invalid status and is not online");
350 return false;
351 }
352 update.setAvailableOnline(availableOnline);
353 mDownloads.put(update.getDownloadId(), new DownloadEntry(update));
354 return true;
355 }

357 @SuppressLint("WakelockTimeout")
358 public void startDownload(String downloadId) {
359 Log.d(TAG, "Starting " + downloadId);
360 if (!mDownloads.containsKey(downloadId) || isDownloading(downloadId)) {
361 return;
362 }
363 DownloadEntry entry = mDownloads.get(downloadId);
364 if (entry == null) {
365 Log.e(TAG, "Could not get download entry");
366 return;
367 }
368 Update update = entry.mUpdate;
369 File destination = new File(mDownloadRoot, update.getName());
370 if (destination.exists()) {
371 destination = Utils.appendSequentialNumber(destination);
372 Log.d(TAG, "Changing name with " + destination.getName());
373 }
374 update.setFile(destination);
375 DownloadClient downloadClient;
376 try {
377 downloadClient = new DownloadClient.Builder()
378 .setUrl(update.getDownloadUrl())
379 .setDestination(update.getFile())
380 .setDownloadCallback(getDownloadCallback(downloadId))
381 .setProgressListener(getProgressListener(downloadId))
382 .setUseDuplicateLinks(true)
383 .build();
384 } catch (IOException exception) {
385 Log.e(TAG, "Could not build download client");
386 update.setStatus(UpdateStatus.PAUSED_ERROR);
387 notifyUpdateChange(downloadId);
388 return;
389 }
390 addDownloadClient(entry, downloadClient);
391 update.setStatus(UpdateStatus.STARTING);
392 notifyUpdateChange(downloadId);
393 downloadClient.start();
394 mWakeLock.acquire();
395 }

397 @SuppressLint("WakelockTimeout")
398 public void resumeDownload(String downloadId) {
399 Log.d(TAG, "Resuming " + downloadId);
400 if (!mDownloads.containsKey(downloadId) || isDownloading(downloadId)) {
401 return;
402 }
403 DownloadEntry entry = mDownloads.get(downloadId);
404 if (entry == null) {
405 Log.e(TAG, "Could not get download entry");
406 return;
407 }
408 Update update = entry.mUpdate;
409 File file = update.getFile();
410 if (file == null || !file.exists()) {
411 Log.e(TAG, "The destination file of " + downloadId + " doesn't exist, can't resume");
412 update.setStatus(UpdateStatus.PAUSED_ERROR);
413 notifyUpdateChange(downloadId);
414 return;
415 }
416 if (file.exists() && update.getFileSize() > 0 && file.length() >= update.getFileSize()) {
417 Log.d(TAG, "File already downloaded, starting verification");
418 update.setStatus(UpdateStatus.VERIFYING);
419 verifyUpdateAsync(downloadId);
420 notifyUpdateChange(downloadId);
421 } else {
422 DownloadClient downloadClient;
423 try {
424 downloadClient = new DownloadClient.Builder()
425 .setUrl(update.getDownloadUrl())
426 .setDestination(update.getFile())
427 .setDownloadCallback(getDownloadCallback(downloadId))
428 .setProgressListener(getProgressListener(downloadId))
429 .setUseDuplicateLinks(true)
430 .build();
431 } catch (IOException exception) {
432 Log.e(TAG, "Could not build download client");
433 update.setStatus(UpdateStatus.PAUSED_ERROR);
434 notifyUpdateChange(downloadId);
435 return;
436 }
437 addDownloadClient(entry, downloadClient);
438 update.setStatus(UpdateStatus.STARTING);
439 notifyUpdateChange(downloadId);
440 downloadClient.resume();
441 mWakeLock.acquire();
442 }
443 }

472 public void deleteUpdate(String downloadId) {
473 Log.d(TAG, "Cancelling " + downloadId);
474 if (!mDownloads.containsKey(downloadId) || isDownloading(downloadId)) {
475 return;
476 }
477 DownloadEntry entry = mDownloads.get(downloadId);
478 if (entry != null) {
479 Update update = entry.mUpdate;
480 update.setStatus(UpdateStatus.DELETED);
481 update.setProgress(0);
482 update.setPersistentStatus(UpdateStatus.Persistent.UNKNOWN);
483 deleteUpdateAsync(update);
484
485 if (!update.getAvailableOnline()) {
486 Log.d(TAG, "Download no longer available online, removing");
487 mDownloads.remove(downloadId);
488 notifyUpdateDelete(downloadId);
489 } else {
490 notifyUpdateChange(downloadId);
491 }
492 }
